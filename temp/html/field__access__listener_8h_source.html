43 // A default/no-op implementation of message hooks.
44 //
45 // See go/statically-dispatched-message-hooks for details.
46 template <typename Proto>
47 struct NoOpAccessListener {
48  // Number of fields are provided at compile time for the trackers to be able
49  // to have stack allocated bitmaps for the fields. This is useful for
50  // performance critical trackers. This is also to avoid cyclic dependencies
51  // if the number of fields is needed.
52  static constexpr int kFields = Proto::_kInternalFieldNumber;
53  // Default constructor is called during the static global initialization of
54  // the program.
55  // We provide a pointer to extract the name of the proto not to get cyclic
56  // dependencies on GetDescriptor() and OnGetMetadata() calls. If you want
57  // to differentiate the protos during the runtime before the start of the
58  // program, use this functor to get its name. We either way need it for
59  // LITE_RUNTIME protos as they don't have descriptors at all.
60  explicit NoOpAccessListener(StringPiece (*name_extractor)()) {}
61  // called repeatedly during serialization/deserialization/ByteSize of
62  // Reflection as:
63  // AccessListener<MessageT>::OnSerialize(this);
64  static void OnSerialize(const MessageLite* msg) {}
65  static void OnDeserialize(const MessageLite* msg) {}
66  static void OnByteSize(const MessageLite* msg) {}
67  static void OnMergeFrom(const MessageLite* to, const MessageLite* from) {}
68 
69  // NOTE: This function can be called pre-main. Make sure it does not make
70  // the state of the listener invalid.
71  static void OnGetMetadata() {}
72 
73  // called from accessors as:
74  // AccessListener<MessageT>::On$operation(this, &field_storage_);
75  // If you need to override this with type, in your hook implementation
76  // introduce
77  // template <int kFieldNum, typename T>
78  // static void On$operation(const MessageLite* msg,
79  // const T* field) {}
80  // And overloads for std::nullptr_t for incomplete types such as Messages,
81  // Maps. Extract them using reflection if you need. Consequently, second
82  // argument can be null pointer.
83  // For an example, see proto_hooks/testing/memory_test_field_listener.h
84  // And argument template deduction will deduce the type itself without
85  // changing the generated code.
86 
87  // add_<field>(f)
88  template <int kFieldNum>
89  static void OnAdd(const MessageLite* msg, const void* field) {}
90 
91  // add_<field>()
92  template <int kFieldNum>
93  static void OnAddMutable(const MessageLite* msg, const void* field) {}
94 
95  // <field>() and <repeated_field>(i)
96  template <int kFieldNum>
97  static void OnGet(const MessageLite* msg, const void* field) {}
98 
99  // clear_<field>()
100  template <int kFieldNum>
101  static void OnClear(const MessageLite* msg, const void* field) {}
102 
103  // has_<field>()
104  template <int kFieldNum>
105  static void OnHas(const MessageLite* msg, const void* field) {}
106 
107  // <repeated_field>()
108  template <int kFieldNum>
109  static void OnList(const MessageLite* msg, const void* field) {}
110 
111  // mutable_<field>()
112  template <int kFieldNum>
113  static void OnMutable(const MessageLite* msg, const void* field) {}
114 
115  // mutable_<repeated_field>()
116  template <int kFieldNum>
117  static void OnMutableList(const MessageLite* msg, const void* field) {}
118 
119  // release_<field>()
120  template <int kFieldNum>
121  static void OnRelease(const MessageLite* msg, const void* field) {}
122 
123  // set_<field>() and set_<repeated_field>(i)
124  template <int kFieldNum>
125  static void OnSet(const MessageLite* msg, const void* field) {}
126 
127  // <repeated_field>_size()
128  template <int kFieldNum>
129  static void OnSize(const MessageLite* msg, const void* field) {}
130 
131  static void OnHasExtension(const MessageLite* msg, int extension_tag,
132  const void* field) {}
